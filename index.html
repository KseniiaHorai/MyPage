function calculateTeamFinanceReport(salaries, team) {
    const salariesLength = Object.keys(salaries).length;
    const teamLength = Object.keys(team).length;

    function getTotalBudgetTeam() {
        let resultArray = { result: 0 };

        let employeesSalaries = {};

        team.forEach((member) => {
            const specialization = member.specialization;
            if (Object.keys(salaries).includes(specialization)) {
                if (
                    !Object.keys(employeesSalaries).includes(
                        `${specialization}`
                    )
                ) {
                    const salaryWithTaxes = salaries[specialization].salary;
                    const budget = calculateBudget(
                        salaryWithTaxes,
                        salaries[specialization].tax
                    );
                    employeesSalaries[specialization] = budget;
                }
                resultArray.result += employeesSalaries[specialization];

                if (!resultArray[specialization]) {
                    resultArray[specialization] = 0;
                }
                resultArray[specialization] +=
                    employeesSalaries[specialization];
            }
        });
        return resultArray;
    }

    function calculateBudget(salaryWithTaxes, tax) {
        tax = tax.slice(0, tax.length - 1);
        const result = (salaryWithTaxes * 100) / (100 - +tax);
        return result;
    }

    if (
        salariesLength >= 1 &&
        salariesLength <= 10 &&
        teamLength >= 1 &&
        teamLength <= 100
    ) {
        const resultArray = getTotalBudgetTeam();
        const result = {
            totalBudgetTeam: Math.trunc(resultArray.result),
        };
        const jobs = Object.keys(salaries);
        jobs.forEach((job) => {
            resultKeyName = `totalBudget${job}`;

            result[resultKeyName] = resultArray[job]
                ? Math.trunc(resultArray[job])
                : 0;
        });
        return result;
    }
    return false;
}
